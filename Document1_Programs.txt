class Main {
  public static void main(String[] args) {
    int a = Integer.parseInt(args[0]);
    int b = Integer.parseInt(args[1]);
    int sum = a+b;
    System.out.println("The Sum of "+ args[0]+ " and "+args[1]+ " is " +sum);
  }
}

Output:
>javac Main.java
>java Main 10 20
The sum of 10 and 20 is 30
